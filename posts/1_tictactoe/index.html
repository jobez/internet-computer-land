async function* renderGame() {
  const state = createState();
  while (true) {
    let body = null;
    const p = new Promise((resolve) => {
      body = renderGrid(state, resolve);
    });

    yield body;
    await p;
  }
}

function cr_to_binary(col, row) {
    var increase;
    switch (col) {
        case 0: 
            increase = 0;
            break;
        case 1:
            increase = 3;
            break;
        case 2:
            increase = 6
;

    }
    
    var binary_idx = row + increase;
    return Math.pow(2, binary_idx);
}
function handleTurn(state, col, row) {
  if (state.board[col][row] === 0) {
    if (state.turn % 2 === 0) {
      state.board[col][row] = X_PLAYER;
      state.x_player_state = state.x_player_state + cr_to_binary(col, row)
    } else {
      state.board[col][row] = O_PLAYER;
      state.o_player_state = state.o_player_state + cr_to_binary(col, row)      
    }
    state.turn++;
    // winGame(state);
    console.log(state, col, row);
  }
}
